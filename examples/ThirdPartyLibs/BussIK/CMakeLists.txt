SET(BussIK_SRCS
	Jacobian.cpp
	LinearR2.cpp
	LinearR3.cpp
	LinearR4.cpp
	MatrixRmn.cpp
	Misc.cpp
	Node.cpp
	Tree.cpp
	VectorRn.cpp
)

SET(BussIK_HDRS
	Jacobian.h
	LinearR2.h
	LinearR3.h
	LinearR4.h
	MathMisc.h
	MatrixRmn.h
	Node.h
	Spherical.h
	Tree.h

	RUNTIME DESTINATION ${BULLET_INSTALL_BINDIR}

	ARCHIVE DESTINATION ${BULLET_INSTALL_LIBDIR})
